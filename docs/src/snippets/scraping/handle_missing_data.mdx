from pydantic import BaseModel


class Product(BaseModel):
    name: str
    price: float
    discount_price: float | None = None  # Optional
    rating: float | None = None  # Optional

import { z } from 'zod';

  discount_price: z.number().nullable().optional(), // Optional

});
